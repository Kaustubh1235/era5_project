def make_sequences(data_x: np.ndarray,
                   data_y: np.ndarray,
                   steps: int = 10):
    """
    Convert 2-D time-series arrays into overlapping sequences.

    Parameters
    ----------
    data_x : ndarray, shape (T, features)
        The full timeline of input features.
    data_y : ndarray, shape (T, 1)
        The corresponding targets (single column).
    steps  : int
        How many past time-steps the model should see.

    Returns
    -------
    X_seq : ndarray, shape (samples, steps, features)
    y_seq : ndarray, shape (samples, 1)
    """
    X_seq, y_seq = [], []
    for i in range(len(data_x) - steps):
        X_seq.append(data_x[i:i+steps])   # window of `steps`
        y_seq.append(data_y[i+steps])     # next value
    return np.asarray(X_seq), np.asarray(y_seq)
